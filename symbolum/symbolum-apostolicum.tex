\subsection*{S\'ymbolum Apost\'olicum}

\begin{verse}
        Credo in unum Deum
            Patrem omnipotentem,
            Creatorem \ul{cæ}li et terræ,
%
\\!     et in Jesum Christum,
            Filium E\ul{i}us \'unicum,
            D\'ominum nostrum,
\\      qui con\ul{ce}ptus est de Sp\'iritu Sancto,
            natus ex Maria V\'ir\ul{gi}ne,
\\      passus sub Pon\ul{ti}o Pilato,
            cru\ul{ci}fixus, mortuus, et sepultus,
\\      de\ul{sc}endit ad \'inferos,
            ter\ul{ti}a die re\ul{s}urrexit a mortuis,
\\      a\ul{sc}endit ad \ul{cæ}los,
            sedet ad d\'exteram Dei Patris omnipotentis,
\\      inde venturus est iudicare
            vivos et mortuos.
%
\\!     Credo in Sp\'iritum Sanctum,
\\      sanctam Eccle\ul{s}iam cath\'olicam,
            sanctorum communionem,
\\      remissionem peccatorum,
\\      carnis re\ul{s}urrec\ul{ti}onem,
\\      vitam æternam.
\end{verse}
