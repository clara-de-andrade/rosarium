\subsection*{S\'ymbolum Nicæum-Constantinopolitanum}

\begin{verse}
        Credo in unum Deum, Patrem omnipotentem,
            factorem cæli et terræ,
\\\vin      visibilium ómnium et invisibilium
%
\\!     Et in unum D\'ominum, Iesum Christum,
            Filium Dei Unig\'enitum,
\\\vin      et ex Patre natum ante \'omnia s\'{æ}cula.
\\\vin      Deum de Deo, lumen de l\'umine,
            Deum verum de Deo vero,
\\\vin      g\'enitum, non factum, consubstantialem Patri:
\\\vin      per quem \'omnia facta sunt.
\\      Qui propter nos h\'omines,
        et propter nostram salutem descendit de cælis.
\\\vin      Et incarnatus est de Sp\'iritu Sancto
            ex Maria V\'irgine,
\\\vin      et homo factus est.
\\      Crucifixus etiam pro nobis sub Pontio Pilato;
            passus et sepultus est,
\\      et resurrexit tertia die, secundum Scripturas,
\\      et ascendit in cælum, sedet ad d\'exteram Patris.
\\      Et \'iterum venturus est cum gloria,
\\\vin      iudicare vivos et mortuos
\\\vin      cuius regni non erit finis.
%
\\!     Et in Spi\'ritum Sanctum, D\'ominum et vivificantem:
\\\vin      qui ex Patre Filioque procedit.
\\\vin      Qui cum Patre et Filio
            simul adoratur et conglorificatur:
\\\vin      qui locutus est per prophetas.
\\      Et unam, sanctam, cath\'olicam et apost\'olicam Ecclesiam.
\\      Confiteor unum baptisma in remissionem peccatorum.
\\      Et exspecto resurrectionem mortuorum,
\\      et vitam venturi s\'{æ}culi.
\end{verse}
