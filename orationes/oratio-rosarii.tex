\subsection*{Oratio Rosarii}
\vspace{-0.5em}

\begin{verse}
        Deus, cuius Unig\'enitus
\\\vin      per vitam, mortem
\\\vin      et resurrectionem
\\      suam nobis salutis æternæ
\\\vin      præmia comparavit,
\\      concede, quæsumus:
\\      ut hæc mysteria
\\\vin      sacrat\'issimo beatæ Mariæ
\\\vin      V\'irginis Rosario recolentes,
\\      et imitemur quod c\'ontinent,
\\      et quod promittunt, assequamur.
\\      Per Christum D\'ominum nostrum.
\end{verse}
